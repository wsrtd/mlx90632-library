47 int mlx90632_start_measurement(void)
48 {
49  int ret, tries = 100;
50  uint16_t reg_status;
51 
52  ret = mlx90632_i2c_read(MLX90632_REG_STATUS, &reg_status);
53  if (ret < 0)
54  return ret;
55 
56  ret = mlx90632_i2c_write(MLX90632_REG_STATUS, reg_status & (~MLX90632_STAT_DATA_RDY));
57  if (ret < 0)
58  return ret;
59 
60  while (tries-- > 0)
61  {
62  ret = mlx90632_i2c_read(MLX90632_REG_STATUS, &reg_status);
63  if (ret < 0)
64  return ret;
65  if (reg_status & MLX90632_STAT_DATA_RDY)
66  break;
67  /* minimum wait time to complete measurement
68  * should be calculated according to refresh rate
69  * atm 10ms - 11ms
70  */
71  usleep(10000, 11000);
72  }
73 
74  if (tries < 0)
75  {
76  // data not ready
77  return -ETIMEDOUT;
78  }
79 
80  return (reg_status & MLX90632_STAT_CYCLE_POS) >> 2;
81 }

419 int32_t mlx90632_init(void)
420 {
421  int32_t ret;
422  uint16_t eeprom_version, reg_status;
423 
424  ret = mlx90632_i2c_read(MLX90632_EE_VERSION, &eeprom_version);
425  if (ret < 0)
426  {
427  return ret;
428  }
429 
430  if ((eeprom_version & 0x00FF) != MLX90632_DSPv5)
431  {
432  // this here can fail because of big/little endian of cpu/i2c
433  return -EPROTONOSUPPORT;
434  }
435 
436  ret = mlx90632_i2c_read(MLX90632_REG_STATUS, &reg_status);
437  if (ret < 0)
438  return ret;
439 
440  // Prepare a clean start with setting NEW_DATA to 0
441  ret = mlx90632_i2c_write(MLX90632_REG_STATUS, reg_status & ~(MLX90632_STAT_DATA_RDY));
442  if (ret < 0)
443  return ret;
444 
445  if ((eeprom_version & 0x7F00) == MLX90632_XTD_RNG_KEY)
446  {
447  return ERANGE;
448  }
449 
450  return 0;
